Two Sum Solution (C++)


class Solution {
public:
    vector twoSum(vector& nums, int target) {
        unordered_map map;
        for (int i = 0; i < nums.size(); i++) {
            int complement = target - nums[i];
            if (map.find(complement) != map.end()) {
                return {map[complement], i};
            }
            map[nums[i]] = i;
        }
        return {};
    }
};
